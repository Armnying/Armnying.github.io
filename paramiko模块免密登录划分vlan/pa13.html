2.4使用Paramiko进行SSH密钥登录并配置VLAN 10

import paramiko
import time
ssh = paramiko.SSHClient()
ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
ssh.connect(hostname='192.168.56.100',
port=22,
username='python',
key_filename=r'C:\Users\Administrator\.ssh\id_rsa') 
cli = ssh.invoke_shell()
cli.send('system-view immediately\n ')
cli.send('vlan 10\n')
time.sleep(2)
cli.send('quit\n')
cli.send('int g1/0/1\n')
cli.send('port link-type access\n')
cli.send('port default vlan 10\n')
cli.send('undo shutdown\n')
cli.send('quit\n')
cli.send('display vlan\n ')
dis_vlan = cli.recv(999999).decode()
print(dis_vlan)
time.sleep(3) 
cli.send('display cu int g1/0/1\n ')
time.sleep(2) 
dis_g101 = cli.recv(999999).decode()
time.sleep(2) 
print(dis_g101)
ssh.close()
